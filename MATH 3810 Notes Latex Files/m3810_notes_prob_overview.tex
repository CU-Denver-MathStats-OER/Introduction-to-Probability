\section*{Introduction:}
Probability is a measure of uncertainty or likelihood of events. There are several approaches to defining and interpreting probability, each with its own advantages and limitations. This document summarizes four common approaches: Classical, Frequentist, Bayesian, and Intuitive.

\section*{Classical (A Priori) Probability:}
	\begin{itemize}
		\item\textbf{Idea:} Probability is based on \emph{symmetry} or \emph{equally likely outcomes}.  

		\item\textbf{Definition:} If an experiment has $n$ equally likely outcomes and an event $A$ contains $k$ of them:
\[P(A) = \frac{k}{n}.\]

		\item\textbf{Example:} Rolling a fair six-sided die. The probability of rolling an even number is
\[P(\text{even}) = \frac{3}{6} = 0.5.\]

		\item\textbf{Advantages:} Simple, intuitive for games of chance.
		
		\item\textbf{Limitations:} Only valid when outcomes are truly equally likely.
		
	\end{itemize}

\section*{Frequentist (Empirical) Probability:}
	\begin{itemize}
		\item\textbf{Idea:} Probability is the \emph{long-run relative frequency} of an event.  

		\item\textbf{Definition:} Repeating an experiment $n$ times, if event $A$ occurs $k_n$ times:
\[P(A) = \lim_{n \to \infty} \frac{k_n}{n}.\]

		\item\textbf{Example:} Flipping a coin 10,000 times and observing 5,020 heads gives
\[P(\text{head}) \approx \frac{5020}{10000} = 0.502.\]

		\item\textbf{Advantages:} Objective, grounded in data.  

		\item\textbf{Limitations:} Requires repeated trials; not suitable for unique events.
	\end{itemize}

\section*{Bayesian Probability:}
	\begin{itemize}
		\item\textbf{Idea:} Probability represents a \emph{degree of belief} and can be updated with new evidence.  

		\item\textbf{Definition:} Given prior belief $p(\theta)$ and data $x$, the posterior is
\[p(\theta \mid x) = \frac{p(x \mid \theta) \, p(\theta)}{p(x)}.\]

		\item\textbf{Example:} Suppose a medical test has sensitivity $95\%$ and specificity $90\%$, and the disease prevalence is $5\%$. The probability that a person who tests positive actually has the disease can be computed using Bayes' theorem:
\[P(\text{disease} \mid \text{positive}) = \frac{0.05 \cdot 0.95}{0.05 \cdot 0.95 + 0.95 \cdot 0.05} \approx 0.50.\]

		\item\textbf{Advantages:} Handles single events, updates with data.  

		\item\textbf{Limitations:} Requires a prior, which can be subjective.
	\end{itemize}
	
\section*{Intuitive / Subjective Probability:}
	\begin{itemize}
		\item\textbf{Idea:} Probability represents a \emph{personal degree of belief} or intuition.  

		\item\textbf{Example:} "Based on experience, I estimate a 70\% chance it will rain tomorrow."  

		\item\textbf{Advantages:} Flexible, can incorporate expert knowledge and one-off events.  

		\item\textbf{Limitations:} Highly subjective; may vary between people.
	\end{itemize}
