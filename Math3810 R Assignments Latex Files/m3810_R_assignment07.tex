\section*{Instructions}
Show all reasoning clearly. All simulation results should be reproducible and clearly labeled.
You may use R for all computations.

\section*{Problems}
\begin{enumerate}[leftmargin=*]

\item \textbf{Joint Distribution Simulation}
\begin{enumerate}
\item Simulate 5000 pairs $(X,Y)$ where $X\sim N(0,1)$, $Y\sim N(0,1)$ independent.
\item Plot the scatterplot of $(X,Y)$ and comment on independence.
\item Compute sample covariance and correlation.
\end{enumerate}

\item \textbf{Dependent Variables}
\begin{enumerate}
\item Create $Z = X + Y$. Plot histogram of $Z$.
\item Compute sample mean and variance.
\item Compare with theoretical mean and variance.
\end{enumerate}

\item \textbf{Bivariate Normal}
\begin{enumerate}
\item Simulate 5000 pairs from bivariate normal with $\rho=0.7$.
\item Plot scatterplot and overlay marginal densities.
\item Compute sample correlation.
\end{enumerate}

\item \textbf{Conditional Probability}
\begin{enumerate}
\item Estimate $P(Y>1 \mid X>0)$ from simulations.
\item Compare with theoretical conditional probability for independent $X,Y$.
\end{enumerate}

\item \textbf{Discussion}
\begin{itemize}
\item Explain how correlation affects scatterplots.
\item Discuss the effect on variance of the sum $Z=X+Y$.
\end{itemize}

\end{enumerate}

\section*{Solutions}
\begin{enumerate}[leftmargin=*]

\item
\begin{lstlisting}[language=R]
set.seed(123)
X <- rnorm(5000)
Y <- rnorm(5000)
plot(X, Y)
cov(X,Y)
cor(X,Y)
\end{lstlisting}

\item
\begin{lstlisting}[language=R]
Z <- X + Y
hist(Z, prob=TRUE)
mean(Z); var(Z)
\end{lstlisting}

\item
\begin{lstlisting}[language=R]
library(MASS)
Sigma <- matrix(c(1,0.7,0.7,1),2,2)
biv <- mvrnorm(5000, mu=c(0,0), Sigma=Sigma)
plot(biv[,1], biv[,2])
cor(biv[,1], biv[,2])
\end{lstlisting}

\item
\begin{lstlisting}[language=R]
mean(Y[X>0] > 1)
\end{lstlisting}

\item
Correlation affects direction and shape of scatterplot; positive correlation tilts upward.
Variance of sum increases with positive correlation.

\end{enumerate}
